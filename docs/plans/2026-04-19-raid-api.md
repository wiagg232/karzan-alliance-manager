# Raid Score API Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship a token-authenticated Supabase Edge Function (`raid-api`) with three endpoints (`GET /season`, `GET /members`, `PUT /scores`), a hand-written OpenAPI spec, and an admin UI (`ApiTokensManager`) for issuing/revoking bearer tokens.

**Architecture:** A single Edge Function with internal routing handles all three endpoints. Auth validates SHA-256 hashed bearer tokens against an `api_tokens` table (RLS-protected, service role bypasses). Writes to `member_raid_records` are restricted to the current non-archived season and trigger guild median recalc. The admin UI hashes tokens in-browser before storing — plaintext is shown once and discarded.

**Tech Stack:** Deno (Edge Functions), Supabase JS v2, PostgreSQL + RLS, React 19 + TypeScript, TailwindCSS, i18next, `crypto.subtle.digest` (SHA-256), OpenAPI 3.1.

**Spec:** [`docs/superpowers/specs/2026-04-19-raid-api-design.md`](../superpowers/specs/2026-04-19-raid-api-design.md)

```ts
export async function sha256Hex(input: string): Promise<string> {
  const data = new TextEncoder().encode(input);
  const buf = await crypto.subtle.digest("SHA-256", data);
  return Array.from(new Uint8Array(buf))
    .map((b) => b.toString(16).padStart(2, "0"))
    .join("");
}
```

```ts
export function medianFloored(scores: number[]): number {
  const positive = scores.filter((s) => s > 0);
  if (positive.length === 0) return 0;
  const sorted = [...positive].sort((a, b) => a - b);
  const mid = Math.floor(sorted.length / 2);
  const raw = sorted.length % 2 !== 0
    ? sorted[mid]
    : (sorted[mid - 1] + sorted[mid]) / 2;
  return Math.floor(raw);
}
```

```ts
export const corsHeaders = {
  "Access-Control-Allow-Origin": "*",
  "Access-Control-Allow-Headers": "authorization, content-type",
  "Access-Control-Allow-Methods": "GET, PUT, OPTIONS",
};

export function handlePreflight(req: Request): Response | null {
  if (req.method !== "OPTIONS") return null;
  return new Response(null, { status: 204, headers: corsHeaders });
}

export function jsonResponse(body: unknown, status = 200): Response {
  return new Response(JSON.stringify(body), {
    status,
    headers: { ...corsHeaders, "Content-Type": "application/json" },
  });
}

export function errorResponse(message: string, status: number, extra: Record<string, unknown> = {}): Response {
  return jsonResponse({ error: message, ...extra }, status);
}
```

```ts
import { createClient, SupabaseClient } from "https://esm.sh/@supabase/supabase-js@2";
import { sha256Hex } from "./lib/auth.ts";
import { corsHeaders, errorResponse, handlePreflight, jsonResponse } from "./lib/cors.ts";
import { medianFloored } from "./lib/median.ts";

const SUPABASE_URL = Deno.env.get("SUPABASE_URL")!;
const SERVICE_ROLE_KEY = Deno.env.get("SUPABASE_SERVICE_ROLE_KEY")!;

function serviceClient(): SupabaseClient {
  return createClient(SUPABASE_URL, SERVICE_ROLE_KEY, {
    auth: { persistSession: false, autoRefreshToken: false },
  });
}

async function authenticate(req: Request, sb: SupabaseClient): Promise<{ tokenId: string } | Response> {
  const header = req.headers.get("authorization") || req.headers.get("Authorization");
  if (!header || !header.toLowerCase().startsWith("bearer ")) {
    return errorResponse("missing bearer token", 401);
  }
  const token = header.slice(7).trim();
  if (!token) return errorResponse("missing bearer token", 401);

  const hash = await sha256Hex(token);
  const { data, error } = await sb
    .from("api_tokens")
    .select("id")
    .eq("token_hash", hash)
    .eq("is_active", true)
    .maybeSingle();

  if (error) return errorResponse("auth lookup failed", 500);
  if (!data) return errorResponse("invalid token", 401);

  // Fire-and-forget last_used_at update
  sb.from("api_tokens").update({ last_used_at: new Date().toISOString() }).eq("id", data.id).then(() => {});

  return { tokenId: data.id };
}

Deno.serve(async (req) => {
  const preflight = handlePreflight(req);
  if (preflight) return preflight;

  const url = new URL(req.url);
  // Path format: /raid-api/<route>
  const segments = url.pathname.split("/").filter(Boolean);
  const route = segments[segments.length - 1] ?? "";

  const sb = serviceClient();
  const authResult = await authenticate(req, sb);
  if (authResult instanceof Response) return authResult;

  try {
    if (req.method === "GET" && route === "season") return await handleGetSeason(sb);
    if (req.method === "GET" && route === "members") return await handleGetMembers(sb);
    if (req.method === "PUT" && route === "scores") return await handlePutScores(req, sb);
    return errorResponse("not found", 404);
  } catch (err) {
    console.error("raid-api unhandled error:", err);
    return errorResponse("internal server error", 500);
  }
});
```

- [ ] **Step 1: Replace `handlePutScores` with validation pipeline**

Replace the stub with:

```ts
interface ScoreRecord {
  member_id: string;
  score?: number;
  season_note?: string;
}

interface PutScoresBody {
  season_id: string;
  records: ScoreRecord[];
}

function parseBody(raw: unknown): PutScoresBody | string {
  if (typeof raw !== "object" || raw === null) return "body must be a JSON object";
  const { season_id, records } = raw as Record<string, unknown>;
  if (typeof season_id !== "string" || !season_id) return "season_id is required";
  if (!Array.isArray(records)) return "records must be an array";
  for (const r of records) {
    if (typeof r !== "object" || r === null) return "each record must be an object";
    const rec = r as Record<string, unknown>;
    if (typeof rec.member_id !== "string" || !rec.member_id) return "member_id is required in every record";
    if (rec.score !== undefined && typeof rec.score !== "number") return "score must be a number";
    if (rec.season_note !== undefined && typeof rec.season_note !== "string") return "season_note must be a string";
  }
  return { season_id, records: records as ScoreRecord[] };
}

async function handlePutScores(req: Request, sb: SupabaseClient): Promise<Response> {
  let raw: unknown;
  try {
    raw = await req.json();
  } catch {
    return errorResponse("invalid JSON body", 400);
  }

  const parsed = parseBody(raw);
  if (typeof parsed === "string") return errorResponse(parsed, 400);
  const { season_id, records } = parsed;

  // Verify season_id is the current (latest non-archived) season
  const { data: currentSeason, error: seasonErr } = await sb
    .from("raid_seasons")
    .select("id")
    .eq("is_archived", false)
    .order("season_number", { ascending: false })
    .limit(1)
    .maybeSingle();
  if (seasonErr) return errorResponse("failed to verify season", 500);
  if (!currentSeason) return errorResponse("no active season", 404);
  if (String(currentSeason.id) !== String(season_id)) {
    return errorResponse("season_id must be the current season", 403);
  }

  if (records.length === 0) {
    return jsonResponse({ updated: 0, guilds_recalculated: [] });
  }

  // Pre-validate all member_ids exist
  const memberIds = [...new Set(records.map((r) => r.member_id))];
  const { data: foundMembers, error: memErr } = await sb
    .from("members")
    .select("id, guild_id")
    .in("id", memberIds);
  if (memErr) return errorResponse("failed to validate members", 500);

  const foundIds = new Set((foundMembers ?? []).map((m) => m.id));
  const invalid = memberIds.filter((id) => !foundIds.has(id));
  if (invalid.length > 0) {
    return errorResponse("unknown member_id(s)", 400, { invalid_member_ids: invalid });
  }

  // Next task: upsert + median recalc. For now, acknowledge validation passed.
  return jsonResponse({ updated: 0, guilds_recalculated: [], _debug: "validation passed, write pending" });
}
```

```ts
      return errorResponse("failed to recalc guild median", 500);

    const gmIds = (guildMembers ?? []).map((m) => m.id);
    if (gmIds.length === 0) continue;

    const { data: scoresRows, error: srErr } = await sb
      .from("member_raid_records")
      .select("score")
      .eq("season_id", season_id)
      .in("member_id", gmIds);
    if (srErr) {
      console.error("fetch member scores failed:", srErr);
      return errorResponse("failed to recalc guild median", 500);

    const scores = (scoresRows ?? []).map((r: { score: number | null }) => r.score ?? 0);
    const median = medianFloored(scores);

    const { error: gUpsertErr } = await sb
      .from("guild_raid_records")
      .upsert({ season_id, guild_id: guildId, member_score_median: median }, { onConflict: "season_id,guild_id" });
    if (gUpsertErr) {
      console.error("upsert guild_raid_records failed:", gUpsertErr);
      return errorResponse("failed to recalc guild median", 500);
    }

  return jsonResponse({
    updated: rows.length,
    guilds_recalculated: [...affectedGuilds],
```

```tsx
  const data = new TextEncoder().encode(input);
  const buf = await crypto.subtle.digest('SHA-256', data);
  return Array.from(new Uint8Array(buf))
    .map(b => b.toString(16).padStart(2, '0'))
    .join('');
}

function generatePlaintextToken(): string {
  const bytes = new Uint8Array(32);
  crypto.getRandomValues(bytes);
  return Array.from(bytes).map(b => b.toString(16).padStart(2, '0')).join('');

      )}

}
```
